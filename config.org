* My Functions

#+begin_src emacs-lisp
(defun asbf-run-macro-until-line (line-number)
  "Run the last keyboard macro until LINE-NUMBER is reached."
  (interactive "nRun macro until line: ")
  (while (< (line-number-at-pos) line-number)
    (call-last-kbd-macro)))

(defun asbf-yaml-indentation-level (s)
  (if (string-match "^ " s)
      (+ 1 (asbf-yaml-indentation-level (substring s 1)))
    0))

(defun asbf-yaml-current-line ()
  (buffer-substring-no-properties (point-at-bol) (point-at-eol)))

(defun asbf-yaml-clean-string (s)
  (let* ((s (replace-regexp-in-string "^[[:blank:]-:]*" "" s))
         (s (replace-regexp-in-string ":$" "" s)))
    s))

(defun asbf-yaml-not-blank-p (s)
  (string-match "[^[:blank:]]" s))

(defun asbf-yaml-path-to-point ()
  (save-excursion
    (let* ((line (asbf-yaml-current-line))
           (level (asbf-yaml-indentation-level line))
           result)
      (while (> (point) (point-min))
        (beginning-of-line 0)
        (setq line (asbf-yaml-current-line))
        (let ((new-level (asbf-yaml-indentation-level line)))
          (when (and (asbf-yaml-not-blank-p line)
                     (< new-level level))
            (setq level new-level)
            (setq result (push (asbf-yaml-clean-string line) result)))))
      (mapconcat 'identity result " => "))))

(defun asbf-yaml-show-path-to-point ()
  (interactive)
  (message (asbf-yaml-path-to-point)))

(eval-after-load 'yaml-mode
  '(progn
     (define-key yaml-mode-map (kbd "C-x p") 'asbf-yaml-show-path-to-point)))

(defun asbf-wait-for-buffer (buffer-name)
  "Wait for the specified buffer to open."
  (while (not (get-buffer buffer-name))
    (sit-for 0.1)))

(defun asbf-ruby-namespaced ()
  (interactive)
  (let ((buffer (get-buffer "*eldoc*")))
    (when buffer
      (kill-buffer buffer)
      (message "Buffer '%s' killed." "*eldoc*")))
  (eldoc--invoke-strategy t)
  (asbf-wait-for-buffer "*eldoc*")
  (with-current-buffer "*eldoc*"
    (goto-line 1)
    (kill-new (buffer-substring (line-beginning-position) (line-end-position)))
    (kill-buffer "*eldoc*")))

(defun asbf-copy-filename-to-clipboard ()
  "Copy the current buffer file name to the clipboard."
  (interactive)
  (let ((filename (if (equal major-mode 'dired-mode)
                      default-directory
                    (buffer-file-name))))
    (when filename
      (kill-new filename)
      (message "Copied buffer file name '%s' to the clipboard." filename))))

(defun asbf-get-repo-name ()
  "Get magit current repo's name"
  (nth 0
       (last
        (split-string
         (magit-toplevel) "/") 2)))

(defun asbf-file-path ()
  "Get file-path relative to repo"
  (string-join (nthcdr 5 (split-string buffer-file-name "/")) "/"))

(defun asbf-file-in-gh ()
  "Copy a file to its address in master on CA Github"
  (interactive)
  (let ((file-gh-address (format "https://github.com/cultureamp/%s/blob/master/%s#L%s"
                                 (asbf-get-repo-name)
                                 (asbf-file-path)
                                 (line-number-at-pos))))
    (kill-new file-gh-address)
    (message "Copied file's github address '%s' to the clipboard." file-gh-address)))

(defun asbf-ticket-url ()
  "Open the jira ticket related to the current branch"
  (interactive)
  (let* ((jira-prefix "https://cultureamp.atlassian.net/browse/")
         (branch-ticket (nth 0 (split-string (magit-get-current-branch) "/"))))
    (format "%s%s" jira-prefix branch-ticket)))

(defun asbf-insert-ticket-url ()
  "insert jira ticket related to the current branch"
  (interactive)
  (insert (asbf-ticket-url)))

(defun asbf-open-ticket ()
  "Open the jira ticket related to the current branch"
  (interactive)
  (let ((ticket-url (asbf-ticket-url)))
    (open_in_browser ticket-url)
    (message "Opened ticket: %s" ticket-url)))

(defalias 'open_in_browser 'browse-url-default-macosx-browser)

(defun asbf-insert-formatted-date ()
  "Prompt for a date format and insert the resulting formatted date at point."
  (interactive)
  (let* ((formats '(("%Y%m%d" . "YYYYMMDD")
                    ("%Y%m%d%H%M%S" . "YYYYMMDDHHMMSS")
                    ("%Y-%m-%d" . "YYYY-MM-DD")
                    ("%Y-%m-%d %H:%M:%S" . "YYYY-MM-DD HH:MM:SS")
                    ("%d-%m-%Y" . "DD-MM-YYYY")
                    ("%d/%m/%Y" . "DD/MM/YYYY")
                    ("%H:%M:%S" . "HH:MM:SS")))
         (choices (mapcar #'cdr formats))
         (selection (completing-read "Choose date format: " choices))
         (format-string (car (rassoc selection formats)))
         (date (shell-command-to-string (concat "date +\"" format-string "\""))))
    (insert (string-trim date))))

(defun try-my-dabbrev-substring (old)
  (let ((old-fun (symbol-function 'he-dabbrev-search)))
    (fset 'he-dabbrev-search (symbol-function 'my-dabbrev-substring-search))
    (unwind-protect
        (try-expand-dabbrev old)
      (fset 'he-dabbrev-search old-fun))))
(defun my-dabbrev-substring-search (pattern &optional reverse limit)
  (let ((result ())
        (regpat (cond ((not hippie-expand-dabbrev-as-symbol)
                       (concat (regexp-quote pattern) "\\sw+"))
                      ((eq (char-syntax (aref pattern 0)) ?_)
                       (concat (regexp-quote pattern) "\\(\\sw\\|\\s_\\)+"))
                      (t
                       (concat (regexp-quote pattern)
                               "\\(\\sw\\|\\s_\\)+")))))
    (while (and (not result)
                (if reverse
                    (re-search-backward regpat limit t)
                  (re-search-forward regpat limit t)))
      (setq result (buffer-substring-no-properties (save-excursion
                                                     (goto-char (match-beginning 0))
                                                     (skip-syntax-backward "w_")
                                                     (point))
                                                   (match-end 0)))
      (if (he-string-member result he-tried-table t)
          (setq result nil)))
    result))
(defun try-expand-flexible-abbrev (old)
  "Try to complete word using flexible matching.
Flexible matching works by taking the search string and then
interspersing it with a regexp for any character. So, if you try
to do a flexible match for `foo' it will match the word
`findOtherOtter' but also `fixTheBoringOrange' and
`ifthisisboringstopreadingnow'.
The argument OLD has to be nil the first call of this function, and t
for subsequent calls (for further possible completions of the same
string).  It returns t if a new completion is found, nil otherwise."
  (if (not old)
      (progn
        (he-init-string (he-lisp-symbol-beg) (point))
        (if (not (he-string-member he-search-string he-tried-table))
            (setq he-tried-table (cons he-search-string he-tried-table)))
        (setq he-expand-list
              (and (not (equal he-search-string ""))
                   (he-flexible-abbrev-collect he-search-string)))))
  (while (and he-expand-list
              (he-string-member (car he-expand-list) he-tried-table))
    (setq he-expand-list (cdr he-expand-list)))
  (if (null he-expand-list)
      (progn
        (if old (he-reset-string))
        ())
    (progn
      (he-substitute-string (car he-expand-list))
      (setq he-expand-list (cdr he-expand-list))
      t)))
(defun he-flexible-abbrev-collect (str)
  "Find and collect all words that flex-matches STR.
See docstring for `try-expand-flexible-abbrev' for information
about what flexible matching means in this context."
  (let ((collection nil)
        (regexp (he-flexible-abbrev-create-regexp str)))
    (save-excursion
      (goto-char (point-min))
      (while (search-forward-regexp regexp nil t)
        (setq collection (cons (thing-at-point 'word) collection))))
    collection))
(defun he-flexible-abbrev-create-regexp (str)
  "Generate regexp for flexible matching of STR.
See docstring for `try-expand-flexible-abbrev' for information
about what flexible matching means in this context."
  (concat "\\b" (mapconcat (lambda (x) (concat "\\w*" (list x))) str "")
          "\\w*" "\\b"))
